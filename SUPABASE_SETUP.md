## Create table: `whats_app_site_contact`

Run this in Supabase SQL editor:

```sql
create table if not exists public.whats_app_site_contact (
  id bigint generated by default as identity primary key,
  created_at timestamptz not null default now(),
  full_name text not null,
  job_title text not null,
  phone text not null,
  email text not null,
  business_name text not null,
  business_website text,
  business_description text not null,
  challenges text not null,
  running_ads text[] not null,
  ad_budget_range text not null,
  expected_timeline text not null,
  whatsapp_help_most text[] not null,
  consent boolean not null,
  language text,
  source text,
  page_url text
);

-- Required if you want to insert from the browser using anon key:
alter table public.whats_app_site_contact enable row level security;

drop policy if exists "public insert" on public.whats_app_site_contact;
create policy "public insert"
on public.whats_app_site_contact
for insert
to anon
with check (true);
```
